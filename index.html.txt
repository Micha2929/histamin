<script>
let daten=[];

fetch("daten.json")
.then(r=>r.json())
.then(d=>daten=d);

const suche=document.getElementById("suche");
const erg=document.getElementById("ergebnis");

suche.addEventListener("input",()=>{

const q=suche.value.trim().toLowerCase();

if(!q){
erg.innerHTML="";
return;
}

const treffer=daten.filter(x=>
x.lebensmittel.toLowerCase().includes(q) ||
x.synonyme.some(s=>s.toLowerCase().includes(q))
);

if(treffer.length>10){
erg.innerHTML="<div class='card'><p>Bitte verfeinern Sie Ihre Suche.</p></div>";
return;
}

if(treffer.length>=4){
erg.innerHTML="<div class='card'><ul>"+treffer.map(x=>"<li>"+x.lebensmittel+"</li>").join("")+"</ul></div>";
return;
}

erg.innerHTML=treffer.map(x=>`
<div class="card">
<h2>${x.lebensmittel}</h2>
<p><strong>${x.einstufung}</strong></p>
${x.warum?`<h3>Warum diese Einstufung?</h3><p>${x.warum}</p>`:""}
${x.hinweis?`<h3>Hinweis</h3><p>${x.hinweis}</p>`:""}
${x.einkauf?`<h3>Beim Einkauf beachten</h3><p>${x.einkauf}</p>`:""}
${x.wissenswert?`<h3>Wissenswert</h3><p>${x.wissenswert}</p>`:""}
</div>`).join("");

});
</script>
